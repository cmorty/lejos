1 import lejos.nxt.*;
 2 /**
 3  * allows interruption of a rotatin task
 4  * @author Ruger
 5  */
 6 public class RotInterrupt
 7 {
 8     /**
 9      * This method has the logic - rotate 270, return, wait for button
10      * then rotate to 0 - wait for button
11      * Another way to get results to thE lcd
12      */
13     public void go()
14     {
15         System.out.println("Interrupt  \n rotation");
16         Sound.twoBeeps();
17         Button.waitForPress();
18         Motor.A.rotate(720,true);
19         showRotation(1);
20         Motor.A.rotateTo(0,true);
21         showRotation(1);
22     }
23     /** 
24      * helper method - saves repeated code
25      */
26      public void showRotation(int row )
27     {
28         while(Motor.A.isRotating())
29         {
30             LCD.drawInt(Motor.A.getTachoCount(),4, 0, row);
31             if(Button.readButtons()>0) Motor.A.stop();
32         }
33         while(Motor.A.getActualSpeed()>0);
34         LCD.drawInt(Motor.A.getTachoCount(), 4,8,row);
35         Button.waitForPress();
36     }
37     /**
38      * 
39      */
40     public static void main(String[] args)
41     {
42         new RotInterrupt().go();
43     }
44 
45 }
